/**
 * Obtiene la información detallada del usuario logueado.
 */
// Ahora recibe 'nombreSesionActiva' como parámetro
function obtenerDetalleUsuarioCompleto(nombreSesionActiva) {
  if (!nombreSesionActiva) return null;

  // 1. Obtener datos de perfil (Admin/Staff)
  const adminsRaw = obtenerDatosAdminsCrudos();
  const datosUser = adminsRaw.find(u => u[0].toString().trim().toUpperCase() === nombreSesionActiva.trim().toUpperCase());
  
  const esAdmin = datosUser ? (datosUser[2] === true || datosUser[2] === "TRUE") : false;
  const rolTexto = esAdmin ? "ADMINISTRADOR" : "STAFF OPERATIVO";

  // 2. Búsqueda exhaustiva en la hoja de LOGS
  let movimientos = [];
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheetLogs = ss.getSheetByName("LOGS");
    
    if (sheetLogs) {
      // Obtenemos todos los valores para procesar en memoria (más rápido que leer celda por celda)
      const data = sheetLogs.getDataRange().getValues();
      const nombreBusqueda = nombreSesionActiva.trim().toUpperCase();

      // BUCLE INVERSO: Empezamos desde la última fila (data.length - 1) hacia arriba (0)
      for (let i = data.length - 1; i >= 0; i--) {
        const fila = data[i];
        
        // El nombre está en la Columna O (índice 14)
        const usuarioEnFila = fila[14] ? fila[14].toString().trim().toUpperCase() : "";

        if (usuarioEnFila === nombreBusqueda) {
          // Ajuste de +2 horas sobre la marca de tiempo en Columna A (índice 0)
          let fechaOriginal = new Date(fila[0]);
          fechaOriginal.setHours(fechaOriginal.getHours() + 2);
          
          let fechaFormateada = Utilities.formatDate(fechaOriginal, Session.getScriptTimeZone(), "dd/MM HH:mm");
          
          // Guardamos el movimiento (f[3] es la acción en Columna D)
          movimientos.push({
            fecha: fechaFormateada,
            accion: fila[3] 
          });
        }

        // IMPORTANTE: En cuanto recolectamos los 5 más recientes del usuario, rompemos el bucle
        if (movimientos.length === 5) break;
      }
    }
  } catch (e) {
    console.log("Error crítico leyendo LOGS: " + e);
  }

  // Resultado final para el modal
  return {
    nombre: nombreSesionActiva,
    rol: rolTexto,
    esAdmin: esAdmin,
    movimientos: movimientos,
    estacion: "LOCACION BJX"
  };
}

function ordenarLimpiarYResaltarNotas() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName("CUADRE"); 
  if (!hoja) return;

  const filaEncabezados = 1;
  const colEstado = 6;  // F
  const colNotas = 9;   // I
  const ultimaFila = hoja.getLastRow();

  if (ultimaFila <= filaEncabezados) return;

  // ⚠️ SE ELIMINÓ EL .sort() QUE DUPLICABA LOS AUTOS. AHORA ES 100% SEGURO.

  const rangoDatos = hoja.getRange(filaEncabezados + 1, 1, ultimaFila - filaEncabezados, colNotas);
  const valores = rangoDatos.getValues(); 
  const colores = []; 

  for (let i = 0; i < valores.length; i++) {
    const nota = valores[i][colNotas - 1]; 
    const estado = String(valores[i][colEstado - 1]).toUpperCase();
    let colorFondo = null; 

    if (nota && nota.toString().trim() !== "") {
      switch (estado) {
        case "LISTO": colorFondo = "#FFF176"; break;
        case "MANTENIMIENTO": colorFondo = "#EF9A9A"; break;
        case "SUCIO": colorFondo = "#FFF9C4"; break;
        case "VENTA": colorFondo = "#64B5F6"; break;
        case "RETENIDA": colorFondo = "#E57373"; break;
        case "RESGUARDO": colorFondo = "#E0E0E0"; break;
        case "TRASLADO": colorFondo = "#CE93D8"; break;
        case "NO ARRENDABLE": colorFondo = "#B0BEC5"; break;
        default: colorFondo = null;
      }
    }
    colores.push([colorFondo]);
  }

  // Aplicar colores solo a la columna de Notas
  try {
    hoja.getRange(filaEncabezados + 1, colNotas, colores.length, 1).setBackgrounds(colores);
  } catch (e) {
    console.warn("Fallo de color: " + e.message);
  }
}

    sheet.getRange(idNota, 7).setValue(fechaRes);
    
    return "OK";
  } catch(e) { return "ERROR"; }
}

/**
 * 📥 INSERCIÓN INTELIGENTE CON AUTO-COMPLETADO
 */
function insertarUnidadDesdeHTML(objeto) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hojaCuadre = ss.getSheetByName("CUADRE");
  
  try {
    const mvaBuscado = String(objeto.mva).trim().toUpperCase();
    const ultimaFila = hojaCuadre.getLastRow();
    
    // 1. VALIDACIÓN ANTI-DUPLICADOS (El MVA ya está en el patio)
    if (ultimaFila > 1) {
      const mvasExistentes = hojaCuadre.getRange(2, 3, ultimaFila - 1, 1).getValues();
      for (let i = 0; i < mvasExistentes.length; i++) {
        if (String(mvasExistentes[i][0]).trim().toUpperCase() === mvaBuscado) {
           return `La unidad ${mvaBuscado} ya está registrada en el patio.`;
        }
      }
    }

    // 2. BÚSQUEDA EN BASE MAESTRA (INDEX_UNIDADES)
    const baseMaestra = obtenerUnidadesVeloz();
    const datosVehiculo = baseMaestra.find(u => u.mva === mvaBuscado);

    // Si el auto no existe en el corporativo, lo rechazamos para no ensuciar el Cuadre
    if (!datosVehiculo) {
        return `Error: El MVA ${mvaBuscado} no existe en la Base Maestra del sistema. Revisa el número.`;
    }

    // 3. PREPARACIÓN DE DATOS (Mezclando lo que dijo la IA con lo que dice la Base Maestra)
    const categReal = datosVehiculo.categoria || "S/C";
    const modeloReal = datosVehiculo.modelo || "S/M";
    const placasReales = datosVehiculo.placas || "S/P";

    const ahora = new Date();
    const timestampNota = Utilities.formatDate(ahora, Session.getScriptTimeZone(), "(yyyy-MM-dd HH:mm:ss)");
    const fechaIngresoPura = Utilities.formatDate(ahora, Session.getScriptTimeZone(), "yyyy-MM-dd'T'HH:mm:ss");
    
    let notasFinales = "";
    if (objeto.notas && objeto.notas.trim() !== "") {
      notasFinales = `${timestampNota} - ${objeto.notas.trim()} - ${objeto.responsableSesion || "S/A"}`;
    }

    const numeroOrden = obtenerOrdenEstado(objeto.estado);
    
    // 4. INSERCIÓN DE LA FILA PERFECTA
    const nuevaFila = [[
      categReal, 
      modeloReal, 
      mvaBuscado, 
      placasReales, 
      "'" + (objeto.gasolina || "N/A"), 
      objeto.estado, 
      objeto.ubicacion, 
      numeroOrden, 
      notasFinales, 
      "", 
      "LIMBO", 
      fechaIngresoPura
    ]];

    hojaCuadre.getRange(ultimaFila + 1, 1, 1, 12).setValues(nuevaFila);

    // 🔥 5. REGISTROS Y PROCESOS DE FONDO
    actualizarFeedSettings(`IN: ${mvaBuscado} (${modeloReal})`, objeto.responsableSesion);
    registrarLog("IN", `📥 INSERTADO: ${modeloReal} (${mvaBuscado})`, objeto.responsableSesion);
    
    SpreadsheetApp.flush(); 

    try {
        if (typeof ordenarLimpiarYResaltarNotas === "function") ordenarLimpiarYResaltarNotas();
    } catch(e) { console.error("Error en procesos secundarios: " + e.message); }

    marcarUltimaModificacion(objeto.responsableSesion);
    return `EXITO|${modeloReal}|${placasReales}`;
    
  } catch (e) {
    return "Error fatal en el servidor: " + e.toString();
  }
}

/**
 * Ejecuta la baja de múltiples unidades de un solo golpe.
 */
function ejecutarEliminacion(listaMvas, responsableSesion) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("CUADRE");
  
  try {
    const ultimaFila = sheet.getLastRow();
    if (ultimaFila < 2) return "La flota está vacía.";
    
    const datosMvas = sheet.getRange(1, 3, ultimaFila, 1).getValues().flat().map(String);
    const totalAELiminar = listaMvas.length;

    for (let i = datosMvas.length - 1; i >= 0; i--) {
      if (listaMvas.indexOf(datosMvas[i]) !== -1) {
        const modelo = sheet.getRange(i + 1, 2).getValue();
        const infoUnidad = `${modelo} (${datosMvas[i]})`;
        sheet.deleteRow(i + 1);
        
        // 🔥 FEED: Reportar la baja
        actualizarFeedSettings(`BAJA: ${datosMvas[i]}`, responsableSesion);

        let mensajeLog = (totalAELiminar === 1) 
          ? `🗑️ SE ELIMINÓ LA UNIDAD: ${infoUnidad}` 
          : `🗑️ ELIMINACIÓN MASIVA: ${infoUnidad}`;

        registrarLog("BAJA", mensajeLog, responsableSesion);
      }
    }

    SpreadsheetApp.flush(); 
    marcarUltimaModificacion(responsableSesion);
    return "EXITO"; 
  } catch (e) {
    return "Error: " + e.toString();
  }
}

/**
 * Obtiene la lista de nombres de administradores desde la hoja CHECK_ADMINS
 */

function obtenerNombreYpassAdmins() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const hoja = ss.getSheetByName("CHECK_ADMINS");
    
    // Verificamos si la hoja existe para evitar errores
    if (!hoja) return { "ERROR": "Hoja no encontrada" };

    const datos = hoja.getDataRange().getValues();
    const baseDatos = {};
    
    for (let i = 1; i < datos.length; i++) {
      let nombre = datos[i][0];
      let pass = datos[i][1];
      
      if (nombre) {
        // Convertimos a String y limpiamos espacios
        baseDatos[nombre.toString().trim().toUpperCase()] = pass.toString().trim();
      }
    }
    
    Logger.log("Admins cargados: " + Object.keys(baseDatos).length);
    return baseDatos; 
  } catch (e) {
    return { "ERROR": e.toString() };
  }
}

/**
 * Procesa la resolución final tras la validación de identidad
 */
function procesarResolverFinal(admin, pass) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const dataJson = PropertiesService.getScriptProperties().getProperty('temp_resolver');
    
    if (!dataJson) return "ERROR: No hay datos temporales para procesar.";
    const objeto = JSON.parse(dataJson);
    
    // 1. Validación de Admin
    const hojaCheck = ss.getSheetByName("CHECK_ADMINS");
    const admins = hojaCheck.getDataRange().getValues();
    let autorizado = false;
    
    for (var i = 1; i < admins.length; i++) {
      if (admins[i][0] === admin && admins[i][1].toString() === pass.toString() && admins[i][2] === true) {
        autorizado = true; 
        break;
      }
    }
    
    if (!autorizado) return "ERROR_AUTH";

    // 2. Preparación de actualización
    const hoja = ss.getSheetByName("CUADREADMINS");
    const fecha = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm");
    
    // Obtenemos la ubicación actual para NO cambiarla (Columna G es la 7)
    const ubicacionActual = hoja.getRange(objeto.fila, 7).getValue();
    
    // Definimos la nueva nota (REEMPLAZA a la anterior)
    const nuevasNotas = "[RESUELTO POR " + admin + " EL " + fecha + "]";

    // 3. Escritura de datos en la fila correspondiente
    // Col F(6): LISTO, G(7): ubicacionActual, H(8): admin, I(9): nuevasNotas, J(10): status, K(11): url, L(12): RESUELTO
    hoja.getRange(objeto.fila, 6, 1, 7).setValues([[ 
      "LISTO", 
      ubicacionActual, 
      admin, 
      nuevasNotas, 
      objeto.archivoStatus, 
      objeto.urlEvidencia, 
      "RESUELTO" 
    ]]);

    // 4. Formato: Pintar la celda de notas (Columna 9 / I) de verde opaco (#d9ead3)
    hoja.getRange(objeto.fila, 9).setBackground("#d9ead3");

    // Limpieza de propiedad temporal
    PropertiesService.getScriptProperties().deleteProperty('temp_resolver');
    return "EXITO";

  } catch (e) { 
    return "ERROR: " + e.toString(); 
  }
}

function procesarFormulario(datos) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName("CUADRE DE FLOTA"); // Asegúrate que el nombre coincida
    
    let urlArchivo = "Sin archivo";
    
    // 1. Manejo del Archivo en Drive
    if (datos.archivo.length > 0) {
      const carpetaId = "ID_DE_TU_CARPETA_AQUÍ"; // Reemplaza con el ID de tu carpeta de Drive
      const carpeta = DriveApp.getFolderById(carpetaId);
      const blob = datos.archivo;
      const archivoSubido = carpeta.createFile(blob);
      urlArchivo = archivoSubido.getUrl();
    }

    // 2. Insertar en la Tabla (Ajusta el orden de las columnas según tu hoja)
    // Suponiendo: CATEG, MODELO, MVA, PLACAS, GASOLINA, ESTADO, UBICACION, NOTAS, ENLACE_ARCHIVO
    sheet.appendRow([
      datos.categ,
      datos.modelo,
      datos.mva,
      datos.placas,
      datos.gasolina,
      datos.estado,
      datos.ubicacion,
      datos.notas,
      urlArchivo
    ]);

    return "¡Unidad registrada con éxito!";
  } catch (e) {
    return "Error: " + e.toString();
  }
}

}


/**
 * Ejecuta la acción final (MOD o DEL) tras la validación de seguridad.
 * El campo 'autor' se llena con el nombre del administrador que validó.
 */
function solicitarValidacionModif(data, tipo) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName("CUADREADMINS");
  const fila = parseInt(data.fila);
  if (!fila || fila < 2) return "ERROR: Fila no válida.";

  try {
    if (tipo === 'MOD') {
      // --- A. GESTIÓN DE DRIVE ---
      let urlFinal = data.urlActual || ""; 
      let tieneArchivo = urlFinal !== "" ? "SI" : "NO"; // Verificamos si ya tenía antes
      let idsNuevos = [];
      
      if (data.archivos && data.archivos.length > 0) {
        var folders = DriveApp.getFoldersByName("ADMINS_BJX2026");
        var folderRaiz = folders.hasNext() ? folders.next() : DriveApp.createFolder("ADMINS_BJX2026");
        
        data.archivos.forEach(function(f) {
          var blob = Utilities.newBlob(Utilities.base64Decode(f.base64), f.mimeType, f.fileName);
          idsNuevos.push(folderRaiz.createFile(blob).getId());
        });

        tieneArchivo = "SI"; // Si se subieron archivos nuevos, forzamos el SI

        if ((urlFinal !== "" && idsNuevos.length > 0) || idsNuevos.length > 1) {
          var nuevaCarpeta = folderRaiz.createFolder("Unidad_" + data.mva + "_MOD");
          if (urlFinal !== "" && !urlFinal.includes("folders")) {
             try {
               var idAnterior = urlFinal.split("/d/")[1].split("/")[0];
               DriveApp.getFileById(idAnterior).moveTo(nuevaCarpeta);
             } catch(e) {}
          }
          idsNuevos.forEach(function(id) { DriveApp.getFileById(id).moveTo(nuevaCarpeta); });
          urlFinal = nuevaCarpeta.getUrl();
        } else if (idsNuevos.length === 1 && urlFinal === "") {
          urlFinal = DriveApp.getFileById(idsNuevos[0]).getUrl();
        }
      }

      // --- B. LÓGICA DE NOTAS ---
      let notaOriginal = hoja.getRange(fila, 9).getValue().toString().trim();
      let notaEntrada = data.notas ? data.notas.trim() : "";
      let notaFinal = notaOriginal;
      const fechaStr = Utilities.formatDate(new Date(), "GMT-6", "(yyyy-MM-dd HH:mm:ss)");

      if (data.borrarNotas === true) {
        // Si marcó el checkbox, la nota queda vacía o con lo nuevo si escribió algo
        notaFinal = (notaEntrada !== "") ? `${fechaStr} ${notaEntrada.toUpperCase()}` : "";
      } else if (notaEntrada !== notaOriginal) {
        // SI HUBO CAMBIO: Reemplazo total con sello de fecha
        notaFinal = (notaEntrada !== "") ? `${fechaStr} ${notaEntrada.toUpperCase()}` : "";
      }

      // --- C. ACTUALIZACIÓN DE COLUMNAS ---
      hoja.getRange(fila, 1).setValue(data.categ);
      hoja.getRange(fila, 2).setValue(data.modelo);
      hoja.getRange(fila, 5).setValue("'" + data.gasolina);
      hoja.getRange(fila, 6).setValue(data.estado);
      hoja.getRange(fila, 7).setValue(data.ubicacion);
      hoja.getRange(fila, 9).setValue(notaFinal);
      
      // COLUMNA J (10): Indicador "SI/NO"
      hoja.getRange(fila, 10).setValue(tieneArchivo);
      
      // COLUMNA K (11): Link de Drive
      if (urlFinal !== "") {
        hoja.getRange(fila, 11).setValue(urlFinal); 
      }

    } else if (tipo === 'DEL') {
      hoja.deleteRow(fila);
    }

    return "OK";
  } catch (e) {
    return "ERROR: " + e.toString();
  }
}

/**
 * Actualiza el estado de la unidad, blindado con LockService y limitado a 5 columnas.
 */
function aplicarEstado(mva, estado, ubi, gas, notasFormulario, borrarNotas, nombreAutor, responsableSesion) {
  const lock = LockService.getScriptLock(); // 🛡️ LLAMAMOS AL CADENERO
  try {
    // Si la base de datos está ocupada por el mapa u otro usuario, esperamos hasta 5 segundos
    lock.waitLock(5000);

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName("CUADRE");
    PropertiesService.getScriptProperties().setProperty('EJECUTANDO_BOTON', 'true');

    const dataRaw = sheet.getDataRange().getValues(); 
    let filaIndex = -1;

    for (let i = 1; i < dataRaw.length; i++) {
      if (dataRaw[i][2].toString() === mva.toString()) {
        filaIndex = i + 1;
        let notaOriginal = dataRaw[i][8] ? dataRaw[i][8].toString().trim() : "";
        let notaEntrada = notasFormulario ? notasFormulario.trim() : "";
        let notaFinal = notaOriginal;
        
        const fechaStr = Utilities.formatDate(new Date(), "GMT-6", "(yyyy-MM-dd HH:mm:ss)");
        const autorStr = nombreAutor ? ` [${nombreAutor}]` : "";

        if (borrarNotas === true || borrarNotas === "true") {
          // Si marcó "Borrar historial", vaciamos la celda. 
          // PERO si escribió algo nuevo, lo guardamos con un sello completamente fresco.
          notaFinal = notaEntrada !== "" ? `${fechaStr}${autorStr} ${notaEntrada}` : "";
          
        } else if (notaEntrada !== notaOriginal) {
          // Si editó el texto manualmente, revisamos usando Regex si borró el sello de fecha "(202X-XX-XX)"
          const tieneSello = /\(\d{4}-\d{2}-\d{2}/.test(notaEntrada);
          
          if (notaEntrada !== "" && !tieneSello) {
            // Si el texto nuevo ya no tiene el sello viejo, se lo inyectamos a la fuerza
            notaFinal = `${fechaStr}${autorStr} ${notaEntrada}`;
          } else {
            // Si todavía conserva el sello viejo, guardamos el texto tal cual
            notaFinal = notaEntrada;
          }
        }

        const nuevoOrden = obtenerOrdenEstado(estado);
        
        // 🔥 CIRUGÍA DE PRECISIÓN 2: Solo agarramos 5 columnas (Gas, Est, Ubi, Orden, Notas).
        // De esta forma NO TOCAMOS la columna K del mapa.
        const filaActualizada = [["'" + gas, estado, ubi, nuevoOrden, notaFinal]];
        sheet.getRange(filaIndex, 5, 1, 5).setValues(filaActualizada);
        
        SpreadsheetApp.flush();
        PropertiesService.getScriptProperties().setProperty('EJECUTANDO_BOTON', 'false');

        // 🔥 FEED: Empujar el movimiento al carrusel de SETTINGS_
        actualizarFeedSettings(`${mva} ➜ ${estado} (${ubi})`, responsableSesion);

        try {
          if (typeof ordenarLimpiarYResaltarNotas === "function") ordenarLimpiarYResaltarNotas();
          if (typeof registrarLog === "function") registrarLog("MODIF", `🔄 MODIFICACION: *${mva}* ESTADO ➜ ${estado} ⛽ GAS ➜ ${gas} 📍 UBI ➜ ${ubi}`, responsableSesion);
        } catch(e) { console.warn("Mantenimiento post-guardado lento."); }

        marcarUltimaModificacion(responsableSesion);
        return "EXITO";
      }
    }
    
    // Si llega aquí, es que no encontró el auto
    PropertiesService.getScriptProperties().setProperty('EJECUTANDO_BOTON', 'false');
    return "ERROR: MVA no encontrado";

  } catch (e) {
    // Si el candado no nos dejó pasar en 5 segundos porque había mucho tráfico
    PropertiesService.getScriptProperties().setProperty('EJECUTANDO_BOTON', 'false');
    return "ERROR: El sistema está muy ocupado, intenta en 2 segundos.";
  } finally {
    lock.releaseLock(); // 🛡️ LIBERAMOS LA PUERTA (PASE LO QUE PASE)
  }
}


/**
 * Retorna el peso jerárquico para el ordenamiento
 */
function obtenerOrdenEstado(estado) {
  const orden = {
    'LISTO': 1,
    'SUCIO': 2,
    'MANTENIMIENTO': 3,
      'RESGUARDO': 4,
    'TRASLADO': 5,
    'NO ARRENDABLE': 6,
    'RETENIDA': 92,
    'VENTA': 93
  };
  return orden[estado.toUpperCase()] || 99;
}

      hojaCuadre.appendRow([
        objeto.categoria, objeto.modelo, objeto.mva, objeto.placas,

/**
 * Función puente: Sube archivos, guarda datos temporales y abre el modal de Admin
 */
function abrirConfirmacionAdmin(data, esResolucion) {
  try {
    var urlFinal = "";
    var tieneArchivo = "NO";
    var folderRaizName = "ADMINS_BJX2026";

    // 1. GESTIÓN DE DRIVE (Múltiples archivos)
    if (data.archivos && data.archivos.length > 0) {
      var folders = DriveApp.getFoldersByName(folderRaizName);
      var folderRaiz = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderRaizName);
      var destinoFinal = folderRaiz;

      // Si son varios archivos, creamos la carpeta Unidad_(MVA)
      if (data.archivos.length > 1) {
        destinoFinal = folderRaiz.createFolder("Unidad_(" + data.mva + ")");
        urlFinal = destinoFinal.getUrl();
      }

      data.archivos.forEach(function(f) {
        var blob = Utilities.newBlob(Utilities.base64Decode(f.base64), f.mimeType, f.fileName);
        var archivo = destinoFinal.createFile(blob);
        // Si es solo 1, el link es directo al archivo
        if (data.archivos.length === 1) urlFinal = archivo.getUrl();
      });
      tieneArchivo = "SI";
    }

    // 2. PREPARAR OBJETO PARA GUARDADO FINAL
    var objetoMemoria = {
      fila: data.fila || null,
      mva: data.mva,
      placas: data.placas,
      modelo: data.modelo,
      categoria: data.categoria,
      gasolina: data.gasolina,
      estado: data.estado,
      ubicacion: data.ubicacion,
      notas: data.notas,
      urlEvidencia: urlFinal,
      archivoStatus: tieneArchivo,
      tipoAccion: esResolucion ? "RESOLVER" : "INSERTAR"
    };

    // Guardamos en la memoria del servidor para procesarInsercionFinal
    PropertiesService.getScriptProperties().setProperty('temp_unidad', JSON.stringify(objetoMemoria));

    // 3. ABRIR EL MODAL (Cambiado de 'ConfirmarAdmin' a 'Autorizar')
    var html = HtmlService.createHtmlOutputFromFile('Autorizar')
        .setWidth(450)
        .setHeight(580);
    SpreadsheetApp.getUi().showModalDialog(html, "Verificación de Seguridad");

    // RETORNO DE ÉXITO: Esto evita el error "undefined"
    return { status: "success" };

  } catch (e) {
    return { status: "error", message: e.toString() };
  }
}

      let mva = fila[idx.mva].trim();
      let cat = fila[idx.categ].trim().toUpperCase() || "S/C";
      let mod = fila[idx.modelo].trim() || "S/M";

      if (ubic === "PATIO" || est === "RESGUARDO" || est === "TRASLADO") {
        if (ubic === "PATIO" && est !== "TRASLADO") patio.total++;
        actualizarObjeto(patio.lista, est, cat, mod, null, orden);
      } 
      
      if (ubic !== "PATIO" || est === "TRASLADO" || est === "RESGUARDO") {
        fuera.total++;
        actualizarObjeto(fuera.lista, est, cat, mod, mva, orden);
      }
    }

    const preparar = (obj) => Object.keys(obj).map(k => ({ nombre: k, ...obj[k] })).sort((a,b) => a.orden - b.orden);

    return { 
      patio: { lista: preparar(patio.lista), total: patio.total },
      fuera: { lista: preparar(fuera.lista), total: fuera.total }
    };
  } catch (e) { return { error: e.toString() }; }
}

function actualizarObjeto(lista, est, cat, mod, mva, ordenMap) {
  if (!lista[est]) lista[est] = { total: 0, categorias: {}, orden: ordenMap[est] || 99 };
  lista[est].total++;
  if (!lista[est].categorias[cat]) lista[est].categorias[cat] = { cant: 0, modelos: [] };
  lista[est].categorias[cat].cant++;
  let identificador = mva ? `${mod} &nbsp; <b style="color:var(--mex-green)">[ ${mva} ]</b>` : mod;
  if (!lista[est].categorias[cat].modelos.includes(identificador)) {
    lista[est].categorias[cat].modelos.push(identificador);
  }
}

// 3. OPTIMIZACIÓN: ORDENAR Y RESALTAR NOTAS (Súper Rápido)
function ordenarLimpiarYResaltarNotas() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName("CUADRE"); 
  const filaEncabezados = 1;
  const colOrden = 8;   // H
  const colEstado = 6;  // F
  const colModelo = 2;  // B
  const colNotas = 9;   // I
  const colMapa = 12;   // K <--- IMPORTANTE: Debemos llegar hasta aquí
  const ultimaFila = hoja.getLastRow();

  if (ultimaFila <= filaEncabezados) return;

  // CORRECCIÓN CLAVE: El rango DEBE incluir la columna 11 (K)
  // para que la ubicación del mapa se mueva junto con el auto al ordenar.
  const rangoDatos = hoja.getRange(filaEncabezados + 1, 1, ultimaFila - filaEncabezados, colMapa);
  
  // Ordenar (Moverá de la A a la K en bloque)
  rangoDatos.sort([
    { column: colOrden, ascending: true }, 
    { column: colModelo, ascending: true }
  ]);

  // Procesamiento de colores (Esto se queda igual, es eficiente)
  const valores = rangoDatos.getValues(); 
  const colores = []; 

  for (let i = 0; i < valores.length; i++) {
    const fila = valores[i];
    const nota = fila[colNotas - 1]; 
    const estado = String(fila[colEstado - 1]).toUpperCase();
    let colorFondo = null; 

    if (nota && nota.toString().trim() !== "") {
      switch (estado) {
        case "LISTO": colorFondo = "#FFF176"; break;
        case "MANTENIMIENTO": colorFondo = "#EF9A9A"; break;
        case "SUCIO": colorFondo = "#FFF9C4"; break;
        case "VENTA": colorFondo = "#64B5F6"; break;
        case "RETENIDA": colorFondo = "#E57373"; break;
        case "RESGUARDO": colorFondo = "#E0E0E0"; break;
        case "TRASLADO": colorFondo = "#CE93D8"; break;
        case "NO ARRENDABLE": colorFondo = "#B0BEC5"; break;
        default: colorFondo = null;
      }
    }
    colores.push([colorFondo]);
  }

  // Escribir los colores solo en la columna I (9)
  hoja.getRange(filaEncabezados + 1, colNotas, colores.length, 1).setBackgrounds(colores);
}

function obtenerMovimientosRecientes() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("LOGS"); // Asegúrate de que este sea el nombre de tu hoja de logs
  const data = sheet.getDataRange().getDisplayValues();
  
  // Si la hoja solo tiene encabezados o está vacía
  if (data.length <= 1) return "Sin movimientos recientes.";

  // 1. Obtener las últimas 5 filas (excluyendo el encabezado)
  // Usamos slice(-5) para tomar las del final de la hoja y luego reverse() 
  // para que el movimiento más nuevo aparezca arriba.
  const ultimosRegistros = data.slice(1).slice(-5).reverse();

  // 2. Procesar cada registro para darle formato
  const listaDeMovimientos = ultimosRegistros.map(fila => {
    const fecha = fila[0];      // Columna A (Fecha/Hora)
    const movimiento = fila[2]; // Columna C (Descripción del movimiento)
    
    // Retornamos una línea limpia con una viñeta y un separador
    // El <span style='color: #888'> pone la hora en gris pequeño para que no estorbe
    return `• ${movimiento} <span style="font-size: 10px; color: #888;">(${fecha})</span>`;
  });

  // 3. Unir todo con el salto de línea de HTML
  return listaDeMovimientos.join('<br>');
}

/**
 * FUNCIÓN DE CONTEO OFICIAL PARA LOG_CUADRE
 * El TOTAL solo suma unidades físicamente en PATIO.
 * RESGUARDO se cuenta aparte y NO suma al TOTAL PATIO.
 */
function obtenerConteoCuadre() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName("CUADRE");
  const datos = hoja.getDataRange().getValues();
  
  let conteo = { 
    LISTOS: 0, 
    SUCIOS: 0, 
    MANTENIMIENTO: 0, 
    RETENIDAS: 0, 
    RESGUARDO: 0, 
    VENTA: 0, 
    TOTAL: 0 // Este será estrictamente el TOTAL EN PATIO
  };

  for (let i = 1; i < datos.length; i++) {
    let estado = datos[i][5]?.toString().toUpperCase().trim();    // Columna F
    let ubicacion = datos[i][6]?.toString().toUpperCase().trim(); // Columna G

    // 1. Contamos RESGUARDO por separado (independiente de ubicación)
    if (estado === "RESGUARDO") {
      conteo.RESGUARDO++;
      // NO SUMA AL TOTAL PATIO
    } 
    // 2. Solo si la ubicación es estrictamente PATIO, procesamos el resto
    else if (ubicacion === "PATIO") {
      
      if (estado === "LISTO") {
        conteo.LISTOS++;
        conteo.TOTAL++;
      } else if (estado === "SUCIO") {
        conteo.SUCIOS++;
        conteo.TOTAL++;
      } else if (estado === "MANTENIMIENTO" || estado === "HYP") {
        conteo.MANTENIMIENTO++;
        conteo.TOTAL++;
      } else if (estado === "RETENIDA") {
        conteo.RETENIDAS++;
        conteo.TOTAL++;
      } else if (estado === "VENTA") {
        conteo.VENTA++;
        conteo.TOTAL++;
      }
    }
  }
  return conteo;
}

/**
 * Obtiene unidades desde INDEX_UNIDADES de forma ultra rápida.
 */
function obtenerUnidadesVeloz() {
  const cache = CacheService.getScriptCache();
  const cached = cache.get("unidades_index_cache");
  
  // 1. Validar caché primero
  if (cached) {
    try {
      return JSON.parse(cached);
    } catch (e) {
      cache.remove("unidades_index_cache"); // Si el JSON está roto, lo borramos
    }
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName("INDEX_UNIDADES");
  if (!hoja) return []; // Error de seguridad: hoja no encontrada

  const ultimaFila = hoja.getLastRow();
  
  // 2. Validar si hay datos (Si solo hay cabecera o está vacía, detenerse)
  if (ultimaFila < 2) return [];

  // 3. Lectura optimizada: especificamos el número exacto de filas
  const filasALeer = ultimaFila - 1;
  const data = hoja.getRange(2, 1, filasALeer, 5).getValues();
  
  const unidades = data.map(r => {
    // Limpieza de datos para evitar valores null/undefined que rompan el JSON
    const modelo = r[1] || "";
    const mva = r[2] || "";
    const placas = r[3] || "";
    
    return {
      sucursal: r[0] || "",
      modelo: modelo,
      mva: mva,
      placas: placas,
      categoria: r[4] || "",
      etiqueta: `${modelo} ${mva} ${placas}`.trim().toUpperCase()
    };
  });

  // 4. Guardar en caché (Máximo 100KB permitidos por Google)
  try {
    const stringData = JSON.stringify(unidades);
    // Si el texto es muy grande, CacheService fallará silenciosamente
    if (stringData.length < 100000) {
      cache.put("unidades_index_cache", stringData, 1500);
    }
  } catch (e) {
    console.warn("Datos demasiado grandes para el caché");
  }

  return unidades;
}

  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const nombreAdmin = datos.adminResponsable || "ADMIN_DESCONOCIDO";
  
  try {
    const mvaBuscado = datos.mva ? datos.mva.toString().toUpperCase().trim() : "";
    let urlGenerada = "";
    if (datos.archivos && datos.archivos.length > 0) urlGenerada = gestionarSubidaArchivos(datos.archivos, mvaBuscado);

    if (tipoAccion === "ADD" || tipoAccion === "INSERTAR") {
      const mvasExistentes = hojaCuadre.getRange(1, 3, hojaCuadre.getLastRow(), 1).getValues().flat().map(String);
      if (mvasExistentes.includes(mvaBuscado)) return "ERROR: La unidad ya existe.";
      const fechaStr = Utilities.formatDate(new Date(), "GMT-6", "(yyyy-MM-dd HH:mm:ss)");
      let notaInicial = datos.notas ? `${fechaStr} [${nombreAdmin}]: ${datos.notas.trim()}` : "";

      const nuevaFila = [
        datos.categ || datos.categoria || "", // A
        datos.modelo || "",                   // B
        mvaBuscado,                           // C
        datos.placas || "",                   // D
        "'" + (datos.gasolina || ""),         // E
        datos.estado || "",                   // F
        datos.ubicacion || "",                // G
        nombreAdmin,                          // H
        notaInicial,                          // I
        urlGenerada ? "SI" : "NO",            // J
        urlGenerada,                          // K
        "PENDIENTE"                           // L
      ];
      hojaCuadre.appendRow(nuevaFila);
    }
    else if (tipoAccion === "DEL" || tipoAccion === "ELIMINAR") {
      const fila = parseInt(datos.fila);
      let filaReal = fila;
      const mvaEnFila = hojaCuadre.getRange(fila, 3).getValue().toString();
      if (mvaEnFila !== mvaBuscado) {
        const mvas = hojaCuadre.getRange(1, 3, hojaCuadre.getLastRow(), 1).getValues().flat().map(String);
        filaReal = mvas.indexOf(mvaBuscado) + 1;
      }
      if (filaReal > 0) hojaCuadre.deleteRow(filaReal);
      else return "ERROR: No se encontró la fila de la unidad.";
    } 
    else if (tipoAccion === "MOD" || tipoAccion === "MODIFICAR") {
      const fila = parseInt(datos.fila);
      const fechaStr = Utilities.formatDate(new Date(), "GMT-6", "(yyyy-MM-dd HH:mm:ss)");
      let notaOriginal = hojaCuadre.getRange(fila, 9).getValue().toString().trim();
      let notaEntrada = datos.notas ? datos.notas.trim() : "";
      let notaFinal = notaOriginal;

      if (datos.borrarNotas === true) {
        notaFinal = (notaEntrada !== "") ? `${fechaStr} [${nombreAdmin}]: ${notaEntrada}` : ""; 
      } else if (notaEntrada !== "") {
        notaFinal = `${fechaStr} [${nombreAdmin}]: ${notaEntrada}\n${notaOriginal}`;
      }

      const valoresNuevos = [[datos.categ, datos.modelo, datos.mva, datos.placas, "'" + datos.gasolina, datos.estado, datos.ubicacion, nombreAdmin, notaFinal]];
      hojaCuadre.getRange(fila, 1, 1, 9).setValues(valoresNuevos);
      
      if (urlGenerada) {
        hojaCuadre.getRange(fila, 10).setValue("SI");
        hojaCuadre.getRange(fila, 11).setValue(urlGenerada);
      }
    }
    SpreadsheetApp.flush(); 
    return "EXITO"; 
  } catch (e) {
    return "ERROR EN SERVIDOR: " + e.toString();
  }
}


function gestionarSubidaArchivos(archivosArray, mva) {
  const rootFolderId = "1V84sAsfAmVWc8yVUXnwvDa6mkimHPDEL"; // Sustituir por el ID real
  const rootFolder = DriveApp.getFolderById(rootFolderId);
  let urlResultado = "";

  // CASO A: Múltiples archivos -> Crear Carpeta específica
  if (archivosArray.length > 1) {
    const nombreCarpeta = `UNIDAD_(${mva})`;
    const nuevaCarpeta = rootFolder.createFolder(nombreCarpeta);
    
    archivosArray.forEach(fileObj => {
      const blob = Utilities.newBlob(Utilities.base64Decode(fileObj.base64), fileObj.mimeType, `${mva}_${fileObj.fileName}`);
      nuevaCarpeta.createFile(blob);
    });
    
    nuevaCarpeta.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    urlResultado = nuevaCarpeta.getUrl();
  } 
  // CASO B: Un solo archivo -> Subir directo a la raíz
  else if (archivosArray.length === 1) {
    const fileObj = archivosArray[0];
    const blob = Utilities.newBlob(Utilities.base64Decode(fileObj.base64), fileObj.mimeType, `${mva}_${fileObj.fileName}`);
    const nuevoArchivo = rootFolder.createFile(blob);
    
    nuevoArchivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    urlResultado = nuevoArchivo.getUrl();
  }

  return urlResultado;
}

// Busca una unidad por MVA en todas las hojas de plazas
function obtenerUnidadesPlazas() {
  const cache = CacheService.getScriptCache();
  const cached = cache.get("unidades_index_cache");
  
  if (cached) {
    try {
      return JSON.parse(cached);
    } catch (e) {
      cache.remove("unidades_index_cache");
    }
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName("INDEX_UNIDADES");
  if (!hoja) return [];

  const ultimaFila = hoja.getLastRow();
  if (ultimaFila < 2) return [];

  // Lectura masiva en una sola operación (Muy veloz)
  const data = hoja.getRange(2, 1, ultimaFila - 1, 5).getValues();
  
  const unidades = data.map(r => {
    const modelo = r[1] || "";
    const mva = r[2] || "";
    const placas = r[3] || "";
    return {
      sucursal: r[0] || "",
      modelo: modelo,
      mva: mva,
      placas: placas,
      categoria: r[4] || "",
      etiqueta: `${modelo} ${mva} ${placas}`.trim().toUpperCase()
    };
  });

  // Intentar guardar en caché solo si el tamaño lo permite
  try {
    const stringData = JSON.stringify(unidades);
    // Google permite máximo 100,000 caracteres en caché
    if (stringData.length < 100000) {
      cache.put("unidades_index_cache", stringData, 1500);
    }
  } catch (e) {
    console.warn("Datos exceden límite de caché, se enviarán sin cachear.");
  }

  return unidades;
}

function obtenerEstructuraMapa() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("MAPA_CONFIG"); 
  const range = sheet.getDataRange();
  const values = range.getValues();
  const mergedRanges = range.getMergedRanges(); // Detecta combinaciones
  const estructura = [];

  for (let r = 0; r < values.length; r++) {
    for (let c = 0; c < values[r].length; c++) {
      let valor = values[r][c];
      
      if (valor !== "") {
        let rowspan = 1;
        let colspan = 1;

        // Verificar si esta celda es el inicio de una combinación
        for (let i = 0; i < mergedRanges.length; i++) {
          let mRange = mergedRanges[i];
          if (mRange.getRow() === r + 1 && mRange.getColumn() === c + 1) {
            rowspan = mRange.getNumRows();
            colspan = mRange.getNumColumns();
            break;
          }
        }

        estructura.push({
          valor: valor,
          row: r + 1,
          col: c + 1,
          rowspan: rowspan,
          colspan: colspan
        });
      }
    }
  }
  return estructura;
}

function obtenerLogsServer() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hoja = ss.getSheetByName('LOGS');
  
  if (!hoja) return [];
  
  const datos = hoja.getDataRange().getValues();
  if (datos.length <= 1) return [];

  const logs = [];
  
  // Recorremos de abajo hacia arriba para traer los más recientes primero
  for (let i = datos.length - 1; i > 0; i--) {
    const fila = datos[i];
    
    // Unimos el texto de las columnas C hasta K (índices 2 al 10)
    let accionCompleta = "";
    for (let col = 2; col <= 10; col++) {
      if (fila[col]) {
        accionCompleta += fila[col] + " ";
      }
    }
    
    // Extraemos la fecha (Columna A)
    let fechaTxt = fila[0];
    if (fechaTxt instanceof Date) {
      fechaTxt = Utilities.formatDate(fechaTxt, "GMT-6", "dd/MM/yyyy HH:mm:ss");
    }

    logs.push({
      fecha: fechaTxt,
      tipo: fila[1] ? fila[1].toString().trim() : "OTRO", // Columna B
      accion: accionCompleta.trim(),                     // Columnas C-K unidas
      autor: fila[13] ? fila[13].toString() : "Sistema"  // Columna N (índice 13)
    });
  }
  
  return logs;
}

function obtenerHistorialLogs() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName("HISTORIAL_PATIO");
    if (!sheet) return [];

    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return []; // Solo están los encabezados

    // Quitamos los encabezados
    data.shift();
    
    // Tomamos solo los últimos 50 movimientos y los invertimos (el más nuevo primero)
    const ultimos = data.slice(-50).reverse();

    return ultimos.map(row => {
      // Formatear la fecha
      let fechaStr = row[0];
      if (fechaStr instanceof Date) {
        fechaStr = Utilities.formatDate(fechaStr, "GMT-6", "dd/MM/yyyy HH:mm:ss");
      }
      
      return {
        fecha: fechaStr || "-",
        accion: (row[1] || "INFO").toString().toUpperCase(), // Ej: MOVE, MODIF
        mva: (row[2] || "S/MVA").toString(),
        // Mostramos el movimiento Ej: "PATIO ➔ L-10"
        detalles: (row[4] || "N/A") + " ➔ " + (row[5] || "N/A"), 
        usuario: (row[6] || "-").toString()
      };
    });
  } catch (e) {
    return [];
  }
}




function modificarUsuario(nombreOriginal, nuevoNombre, nuevoPin, isAdmin) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName("USUARIOS");
    if (!sheet) return "ERROR: No existe la pestaña USUARIOS";

    const rol = isAdmin ? "ADMIN" : "VISITANTE";
    const originalLimpio = nombreOriginal.toString().trim().toUpperCase();
    const nuevoLimpio = nuevoNombre.toString().trim().toUpperCase();
    
    const data = sheet.getDataRange().getValues();
    
    // Si le están cambiando el nombre, verificar que el nuevo nombre no exista ya
    if (originalLimpio !== nuevoLimpio) {
      for (let i = 1; i < data.length; i++) {
        if (data[i][0] && data[i][0].toString().trim().toUpperCase() === nuevoLimpio) {
          return "DUPLICADO";
        }
      }
    }

    // Buscar la fila del usuario original y actualizar sus 3 columnas
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] && data[i][0].toString().trim().toUpperCase() === originalLimpio) {
        // Actualiza de un solo golpe: Nombre, PIN, Rol
        sheet.getRange(i + 1, 1, 1, 3).setValues([[nuevoLimpio, nuevoPin.toString().trim(), rol]]);
        return "EXITO";
      }
    }
    
    return "ERROR: Usuario original no encontrado";
  } catch(e) {
    return "ERROR: " + e.toString();
  }
}

// VERIFICACIÓN DE ADMINISTRADOR GLOBAL
// ==========================================
function verificarAdminGlobal(nombreUsuario) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName("CHECK_ADMINS"); // Asegúrate de que la hoja se llame así
    if (!sheet) return false;

    const data = sheet.getDataRange().getValues();
    const nombreLimpio = nombreUsuario.toString().trim().toUpperCase();
    
    // Recorremos la hoja (omitimos la fila 1 de encabezados)
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] && data[i][0].toString().trim().toUpperCase() === nombreLimpio) {
        // Verificamos si la columna C (índice 2) es TRUE (booleano o texto)
        let isAdminGlobal = data[i][2];
        return isAdminGlobal === true || isAdminGlobal.toString().toUpperCase() === "TRUE";
      }
    }
    return false; // Si no lo encuentra, no es global
  } catch(e) {
    return false; // Por seguridad, ante cualquier error denegamos el acceso
  }
}

// 📡 RADAR: CHEQUEO DE ESTADOS Y NOTIFICACIONES
// ==========================================
function checarNotificaciones(usuarioActivo) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetSettings = ss.getSheetByName("SETTINGS_");
  
  let response = { 
    incidenciasPendientes: 0, 
    alertas: [], 
    mensajesSinLeer: 0, 
    error: null, 
    ultimaActualizacion: "--/-- 00:00",
    ultimoCuadre: "Sin registro",
    mapaBloqueado: false,
    estadoCuadreV3: "LIBRE",
    adminIniciador: "",
    liveFeed: [] 
  };

  try {
    // 1. LECTURA DE INCIDENCIAS
    const sheetInc = ss.getSheetByName("INCIDENCIAS");
    if (sheetInc && sheetInc.getLastRow() > 1) {
      const dataInc = sheetInc.getDataRange().getValues();
      response.incidenciasPendientes = dataInc.filter(r => r[5] === "PENDIENTE").length;
    }

    // 2. LECTURA DE ALERTAS GLOBALES
    const sheetAlt = ss.getSheetByName("ALERTAS");
    if (sheetAlt && sheetAlt.getLastRow() > 1) {
      const dataAlt = sheetAlt.getDataRange().getValues();
      for (let i = 1; i < dataAlt.length; i++) {
        let leidoPor = (dataAlt[i][7] || "").toString();
        if (dataAlt[i][0] && !leidoPor.includes(usuarioActivo)) {
          response.alertas.push({
            id: dataAlt[i][0],
            fecha: (dataAlt[i][1] instanceof Date) ? Utilities.formatDate(dataAlt[i][1], "GMT-6", "dd/MM HH:mm") : dataAlt[i][1],
            autor: dataAlt[i][2], tipo: dataAlt[i][3], titulo: dataAlt[i][4], mensaje: dataAlt[i][5], imagen: dataAlt[i][6]
          });
        }
      }
    }

    // 3. 📩 LECTURA DE MENSAJES PRIVADOS
    const sheetMsg = ss.getSheetByName("MENSAJES");
    if (sheetMsg && sheetMsg.getLastRow() > 1) {
      const dataMsg = sheetMsg.getDataRange().getValues();
      response.mensajesSinLeer = dataMsg.filter(r => r[3] === usuarioActivo && r[5] !== "SI").length;
    }

    // 4. 🔒 LECTURA DE CUADRE Y BLOQUEOS (SETTINGS_)
    if (sheetSettings) {
      const dataF = sheetSettings.getRange("F2:F3").getValues();
      response.estadoCuadreV3 = dataF[0][0] || "LIBRE"; 
      response.adminIniciador = dataF[1][0] || ""; 
      
      const dataFull = sheetSettings.getRange("A2:C12").getValues();
      
      response.ultimaActualizacion = (dataFull[0][0] instanceof Date) ? Utilities.formatDate(dataFull[0][0], "GMT-6", "dd/MM - HH:mm") : dataFull[0][0];
      if (dataFull[1][1]) response.ultimoCuadre = `${dataFull[1][1]} (${dataFull[1][2]})`;
      response.mapaBloqueado = (dataFull[3][1] === true || dataFull[3][1] === "TRUE");

      // 🕒 LECTURA DEL FEED
      for (let i = 6; i <= 10; i++) {
        if (dataFull[i][0]) { 
          let fv = dataFull[i][1];
          let horaTxt = (fv instanceof Date) ? Utilities.formatDate(fv, "America/Mexico_City", "HH:mm") : fv.toString();
          response.liveFeed.push({
            accion: dataFull[i][0].toString(),
            fecha: horaTxt,
            autor: dataFull[i][2].toString()
          });
        }
      }
    }
  } catch(e) { response.error = e.toString(); }
  
  return response;
}

function obtenerTodasLasAlertas() {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("ALERTAS");
    if (!sheet) return [];
    
    const data = sheet.getDataRange().getValues();
    let historial = [];
    
    // Recorremos de abajo hacia arriba (más nuevas primero), ignorando la fila 1
    for (let i = data.length - 1; i > 0; i--) {
      if(data[i][0]) {
        
        // 🛠️ LA CURA: Forzamos la fecha a ser un Texto seguro
        let fechaCruda = data[i][1];
        let fechaStr = "";
        if (fechaCruda instanceof Date) {
          fechaStr = Utilities.formatDate(fechaCruda, "GMT-6", "dd/MM/yyyy HH:mm");
        } else {
          fechaStr = (fechaCruda || "").toString();
        }

        historial.push({
          id: data[i][0],
          fecha: fechaStr, // Enviamos el texto limpio
          autor: data[i][2] || "",
          tipo: data[i][3] || "INFO",
          titulo: data[i][4] || "",
          mensaje: data[i][5] || "",
          leidoPor: data[i][7] || ""
        });
      }
    }
    return historial;
  } catch(e) { return []; }
}

    if (!listaCorreos) return "ERROR: No hay destinatarios";

    // 1. IMAGEN DEL MAPA (Adjunto 1)
    const decodedImage = Utilities.base64Decode(base64Image.split(",")[1]);
    const blobImg = Utilities.newBlob(decodedImage, 'image/png', 'Captura_Mapa.png');

    // 2. OBTENER DESGLOSE CATEGORIZADO Y DATOS PARA LA TABLA DEL PDF
    const dataFlota = sheetCuadre.getDataRange().getDisplayValues();
    let desglose = {};

    // Arrays para ordenar la tabla del PDF

      "RESGUARDO": [], "VENTA": [], "RETENIDA": [], "NO ARRENDABLE": [], "HYP": [], "OTROS": []

    for (let i = 1; i < dataFlota.length; i++) {

      let gas = (dataFlota[i][4] || "N/A").trim();
      let ubi = (dataFlota[i][6] || "SIN UBI").trim();

      // Para el correo (Resumen)
      if(!desglose[estado]) desglose[estado] = { total: 0, cats: {} };
      desglose[estado].total++;
      
      if(!desglose[estado].cats[categ]) desglose[estado].cats[categ] = { total: 0, modelos: {} };
      desglose[estado].cats[categ].total++;
      if(!desglose[estado].cats[categ].modelos[modelo]) desglose[estado].cats[categ].modelos[modelo] = [];

    return "ERROR_EMAIL: " + e.toString();

function guardarAuditoriaCruzada(datosAuditoria, autor) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheetHistory = ss.getSheetByName("HISTORIAL_AUDITORIAS");
  let sheetSettings = ss.getSheetByName("SETTINGS_");

  const cantSobrantes = datosAuditoria.filter(u => u.status === 'EXTRA').length;

  sheetSettings.getRange("A2").setValue(new Date()); // Ping al radar para bloquear botones

  return "EXITO";
}

function obtenerHistorialCuadres() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("HISTORIAL_AUDITORIAS");
  if (!sheet) return [];
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return []; 
  
  const historial = [];
  for (let i = data.length - 1; i > 0; i--) {
    let fechaVal = data[i][0];
    let fechaStr = (fechaVal instanceof Date) ? Utilities.formatDate(fechaVal, "America/Mexico_City", "dd/MM/yyyy HH:mm:ss") : String(fechaVal);

   historial.push({
      fecha: fechaStr,
      auxiliar: String(data[i][1] || "Desconocido"),
      faltantes: data[i][2] || 0,
      sobrantes: data[i][3] || 0,
      ok: data[i][4] || 0,
      pdfUrl: String(data[i][7] || ""),
      admin: String(data[i][8] || "No registrado") // <--- DATA[i][8] LEE LA COLUMNA "I"
    });
  }
  return historial;
}

function marcarMensajesLeidosArray(idsArray) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("MENSAJES");
  if(!sheet || !idsArray || idsArray.length === 0) return;
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (idsArray.includes(data[i][0])) {
      sheet.getRange(i + 1, 6).setValue("SI"); 
    }
  }
  return "OK";
}

function obtenerMensajesPrivados(usuario) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("MENSAJES");
  if (!sheet) return [];
  
  const data = sheet.getDataRange().getValues();
  let inbox = [];
  
  const me = String(usuario).trim().toUpperCase();
  
  for (let i = data.length - 1; i > 0; i--) {
    // 🔥 BLINDAJE SERVIDOR: Leemos limpiando espacios basura de Excel
    const rem = String(data[i][2]).trim().toUpperCase();
    const dest = String(data[i][3]).trim().toUpperCase();
    
    if (dest === me || rem === me) {
      
      let fechaCruda = data[i][1];
      let fechaString = (fechaCruda instanceof Date) ? Utilities.formatDate(fechaCruda, "America/Mexico_City", "dd/MM/yyyy HH:mm") : String(fechaCruda || "");

      inbox.push({
        id: data[i][0],
        fecha: fechaString, 
        remitente: rem,
        destinatario: dest,
        mensaje: data[i][4],
        leido: String(data[i][5]).trim().toUpperCase() === "SI",
        esMio: rem === me 
      });
    }
  }
  return inbox;
}

/**
 * 🛡️ GUARDADO MAESTRO DE POSICIONES (SIN PÉRDIDA DE DATOS)
 */
function guardarNuevasPosiciones(reporte, usuarioResponsable) {
  const lock = LockService.getScriptLock();
  try {
    // Esperamos hasta 10 segundos por el turno (antes eran 5)
    lock.waitLock(10000); 

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const hojasAActualizar = ["CUADRE", "EXTERNOS"];
    const historial = ss.getSheetByName("HISTORIAL_PATIO") || ss.insertSheet("HISTORIAL_PATIO");
    
    // Mapeamos el reporte para búsqueda rápida
    const posicionesNuevas = {};
    reporte.forEach(item => {
      if(item.mva && item.pos) posicionesNuevas[item.mva.toString().trim().toUpperCase()] = item.pos.toUpperCase();
    });

    hojasAActualizar.forEach(nombreHoja => {
      const sheet = ss.getSheetByName(nombreHoja);
      if (!sheet) return;

      // LEEMOS DENTRO DEL LOCK para tener la versión más fresca de la hoja
      const data = sheet.getDataRange().getValues();
      const nuevasPosicionesColK = [];
      let cambiosEnHoja = false;

      for (let i = 0; i < data.length; i++) {
        if (i === 0) {
          nuevasPosicionesColK.push([data[i][10] || "POSICION_MAPA"]);
          continue;
        }

        const mva = data[i][2] ? data[i][2].toString().trim().toUpperCase() : "";
        let posActualEnHoja = data[i][10] ? data[i][10].toString().toUpperCase() : "LIMBO";
        let posPropuesta = posicionesNuevas[mva];

        // Solo si el auto existe en el reporte y su posición es distinta a la de la hoja
        if (mva && posPropuesta && posPropuesta !== posActualEnHoja) {
          historial.appendRow([new Date(), "MOVE", mva, nombreHoja, posActualEnHoja, posPropuesta, usuarioResponsable]);
          posActualEnHoja = posPropuesta;
          cambiosEnHoja = true;
        }
        nuevasPosicionesColK.push([posActualEnHoja]);
      }

      if (cambiosEnHoja) {
        // Escribimos con precisión quirúrgica en la Columna 11 (K)
        sheet.getRange(1, 11, nuevasPosicionesColK.length, 1).setValues(nuevasPosicionesColK);
      }
    });

    SpreadsheetApp.flush(); // Forzamos la escritura inmediata
    return true;

  } catch (e) {
    console.error("Error en autoguardado: " + e.toString());
    return false;
  } finally {
    lock.releaseLock();
  }
}

 * Si no la encuentra, devuelve el URL de 'no-model.png' de la misma carpeta.
 */
function obtenerUrlImagenModelo(modelo) {
  if (!modelo) return "";
  const props = PropertiesService.getScriptProperties();
  const folderId = "1bNgwxrH3h91VfQNox1TVxNmjYJijfd-2"; // Tu carpeta de modelos 
  
  let dbImagenes;
  try {
    dbImagenes = props.getProperty('DB_IMAGENES_MODELOS');
  } catch(e) { dbImagenes = null; }

  // 1. Generamos la base de datos de imágenes si no existe
  if (!dbImagenes || dbImagenes === "{}" || dbImagenes === "") {
    const folder = DriveApp.getFolderById(folderId);
    const files = folder.getFiles();
    const imagesMap = {};
    let placeholderUrl = ""; 

    while(files.hasNext()){
      const file = files.next();
      const nameLower = file.getName().toString().toLowerCase().trim();
      const mime = file.getMimeType();
      
      if(mime.indexOf('image/') === 0) {
        const imgUrl = "https://drive.google.com/thumbnail?id=" + file.getId() + "&sz=w600";
        imagesMap[nameLower] = imgUrl;

        // 🔥 IDENTIFICAMOS EL PLACEHOLDER SOLICITADO
        if (nameLower === "no-model.png") { placeholderUrl = imgUrl; }
      }
    }
    
    if (placeholderUrl) imagesMap["__PLACEHOLDER__"] = placeholderUrl;
    dbImagenes = JSON.stringify(imagesMap);
    props.setProperty('DB_IMAGENES_MODELOS', dbImagenes);
  }
  
  const mapImagenes = JSON.parse(dbImagenes);
  const searchKey = modelo.toString().trim().split(" ")[0].toLowerCase(); 
  
  // 2. Buscamos el modelo
  for (let fileNameLower in mapImagenes) {
     if (fileNameLower.includes(searchKey)) { return mapImagenes[fileNameLower]; }
  }

  // 3. 🔥 SI NO HAY MODELO, DEVOLVEMOS no-model.png 
  return mapImagenes["__PLACEHOLDER__"] || ""; 
}

    const sheetLogs = ss.getSheetByName("HISTORIAL_PATIO") || ss.getSheetByName("LOGS");
    if (sheetLogs) {
      const lastRow = sheetLogs.getLastRow();
      const startRow = Math.max(2, lastRow - 29);
      const count = lastRow - startRow + 1;
      const logs = sheetLogs.getRange(startRow, 1, count, 7).getDisplayValues();
      historialReciente = logs
        .filter(l => l[0] && l[2])
        .map(l => `[${l[0]}] ${l[2]}: de ${l[4]} a ${l[5]} por ${l[6]}`)
        .join("\n");
    }
  } catch (e) { historialReciente = "Sin historial disponible."; }
 
  // ─── DATOS DEL USUARIO EN SESIÓN ──────────────────────────────────────
  let rolUsuario = "operador"; 
  let nombreUsuario = "Operador";
  
  try {
    const ctx = JSON.parse(contextoPatio);
    if (ctx._meta) {
      rolUsuario = (ctx._meta.rol || "operador").toLowerCase();
      nombreUsuario = ctx._meta.nombre || "Operador";
    }
  } catch(e) {}
 
  // ─── CONSTRUIR PROMPT MAESTRO ─────────────────────────────────────────
  const promptMaestro = buildPromptMaestro(
    instruccionUsuario,
    contextoPatio,
    ultimoMVA,
    inventarioGlobal,
    historialReciente,
    fechaHoyStr,
    rolUsuario,
    nombreUsuario
  );
 
  // ─── LLAMAR A GEMINI ──────────────────────────────────────────────────
  const payload = { 
    "contents": [{ "parts": [{ "text": promptMaestro }] }],
    "generationConfig": {
      "temperature": 0.2,
      "topK": 20,
      "topP": 0.8,
      "maxOutputTokens": 2048
    }
  };
  
  const options = { 
    "method": "post", 
    "contentType": "application/json", 
    "payload": JSON.stringify(payload), 
    "muteHttpExceptions": true 
  };
 
  try {
    const response = UrlFetchApp.fetch(url, options);
    const responseData = JSON.parse(response.getContentText());
    
    if (responseData.error) {
      return { 
        "acciones": [{ 
          "accion": "ERROR", 
          "respuesta_voz": "Error de la API: " + (responseData.error.message || "Desconocido"),
          "respuesta_visual": "<div class='ia-alert'>❌ Error API Gemini</div>"
        }] 
      };
    }
    
    let resRaw = responseData.candidates[0].content.parts[0].text;
    resRaw = resRaw.replace(/```json/gi, "").replace(/```/g, "").trim();
    const parsed = JSON.parse(resRaw);
    
    // ── SEGUNDA CAPA DE SEGURIDAD Y AUTOCOMPLETADO (BACKEND) ──────────────
    if (parsed.acciones) {
      
      // 1. CARGAR BASE MAESTRA SOLO SI SE REQUIERE INSERTAR
      let baseMaestra = null;
      const requiereMaestra = parsed.acciones.some(a => a.accion === "INSERTAR_NUEVO");
      if (requiereMaestra) {
        try {
          baseMaestra = obtenerUnidadesVeloz(); // Tu función existente en codigo.gs
        } catch(e) { console.error("Error cargando base maestra:", e); }
      }

      parsed.acciones = parsed.acciones.map(cmd => {
        const accionesBloqueadasOperador = ["ELIMINAR", "INSERTAR_NUEVO"];
        const accionesPermitidasConsulta = ["HABLAR"];
        
        // A) BLOQUEO PARA OPERADOR
        if (rolUsuario === "operador" && accionesBloqueadasOperador.includes(cmd.accion)) {
          return {
            "accion": "HABLAR",
            "mva": cmd.mva || "",
            "respuesta_voz": "Acción bloqueada. No tienes permisos de Administrador para realizar esta operación.",
            "respuesta_visual": "<div class='ia-alert'>⛔ ACCESO DENEGADO<br>Requiere rol de Administrador.</div>"
          };
        }
        
        // B) BLOQUEO PARA CONSULTA
        if (rolUsuario === "consulta" && !accionesPermitidasConsulta.includes(cmd.accion)) {
          return {
            "accion": "HABLAR",
            "mva": cmd.mva || "",
            "respuesta_voz": "Tu perfil es solo de consulta. Contacta a un administrador.",
            "respuesta_visual": "<div class='ia-alert'>🔒 PERFIL CONSULTA<br>Solo puedes consultar información.</div>"
          };
        }

        // C) 🪄 AUTOCOMPLETADO INTELIGENTE PARA INSERTAR_NUEVO
        if (cmd.accion === "INSERTAR_NUEVO" && rolUsuario === "admin") {
          cmd.data = cmd.data || {};
          let modeloReal = cmd.data.modelo || "S/M";
          let placasReales = cmd.data.placas || "S/P";
          let catReal = cmd.data.categoria || "N/A";

          // Buscamos el carro en la Base Maestra
          if (baseMaestra && cmd.mva) {
            const autoMaestro = baseMaestra.find(u => u.mva.toUpperCase() === cmd.mva.toUpperCase());
            if (autoMaestro) {
              modeloReal = autoMaestro.modelo || modeloReal;
              placasReales = autoMaestro.placas || placasReales;
              catReal = autoMaestro.categoria || autoMaestro.categ || catReal;
            }
          }

          // Inyectamos los datos reales para que el frontend los reciba y los inserte en CUADRE
          cmd.data.modelo = modeloReal;
          cmd.data.placas = placasReales;
          cmd.data.categoria = catReal;
          cmd.data.estado = cmd.data.estado || "SUCIO";
          cmd.data.ubicacion = cmd.data.ubicacion || "PATIO";
          cmd.data.gasolina = cmd.data.gasolina || "N/A";
          
          // Mejoramos el diseño visual con los datos verdaderos
          cmd.respuesta_visual = `
            <div class='ia-success' style='text-align:center;'>
              <div style='font-size:10px; letter-spacing:1px; margin-bottom:5px;'>INGRESO EXITOSO</div>
              <div style='font-size:24px; font-weight:900;'>${cmd.mva}</div>
              <div style='font-size:12px; font-weight:700; margin-bottom:8px;'>🚗 ${modeloReal} • 🏷️ ${placasReales}</div>
              <div class='ia-badge-row' style='justify-content:center;'>
                <span class='ia-badge gray'>${cmd.data.ubicacion}</span>
                <span class='ia-badge sucio'>${cmd.data.estado}</span>
              </div>
            </div>`;
        }
        
        return cmd;
      });
    }
    
    return parsed;
    
  } catch (e) {
    return { 
      "acciones": [{ 
        "accion": "ERROR", 
        "respuesta_voz": "Fallo de conexión o sintaxis. Intenta de nuevo.",
        "respuesta_visual": "<div class='ia-alert'>❌ Fallo de Conexión<br>Revisa tu internet o intenta de nuevo.</div>"
      }] 
    };
  }
}
